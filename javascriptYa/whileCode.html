    <script>
      var frase='Todo estará bien', entrada;
      entrada = prompt('Introduzca la frase secreta: ', '');
      while(entrada!=frase)
      {
        entrada = prompt('ERROR \n Intente de nuevo. Introduzca la frase secreta: ', '');
      }
    </script>
    <p>
      Felicidades, ha adivinado la frase secreta.
